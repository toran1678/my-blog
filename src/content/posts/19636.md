```cpp
#include <iostream>
#include <cmath>

using namespace std;

int main() {
  // 다이어트 전 체중 w0
  // 일일 에너지 섭취량 i0
  // 기초 대사량 변화 역치 t
  int w0, i0, t;
  std::cin >> w0 >> i0 >> t;

  // 일일 기초 대사량
  int bmr = i0;

  // 일일 활동 대사량 = 0
  int A0 = 0;
  
  // 다이어트 기간 d
  // 다이어트 기간 일일 섭취량 i
  // 다이어트 기간 활동 대사량 a
  int d, i, a;
  std::cin >> d >> i >> a;

  // 일일 에너지 소비량 = 일일 기초 대사량 + 일일 활동 대사량
  // 체중 = 일일 에너지 섭취량 - (일일 기초 대사량 + 일일 활동 대사량)

  // if (i - (i0 + a) > t)일 경우
  // 일일 기초 대사량 = (i - (i0 + a)) / 2

  int w1 = w0, w2 = w0;

  // 일일 기초 대사량의 변화를 고려하지 않았을 때
  for (int j = 0; j < d; j++) {
    w1 += i - (bmr + a);

    if (w1 <= 0) {
      break;
    }
  }

  if (w1 <= 0) {
    cout << "Danger Diet\n";
  } else {
    cout << w1 << " " << i0 << "\n";
  }

  // 일일 기초 대사량의 변화를 고려했을 때
  for (int j = 0; j < d; j++) {
    // 일일 체중 변화
    w2 += i - (bmr + a);

    // 일일 bmr 변화
    if (abs(i - (bmr + a)) > t) {
      bmr += floor((i - (bmr + a)) / 2.0);
    }

    if (w2 <= 0 || bmr <= 0) {
      break;
    }
  }

  if (w2 <= 0 || bmr <= 0) {
    cout << "Danger Diet\n";
  } else {
    cout << w2 << " " << bmr << " ";
    if (i0 - (bmr + 0) > 0) {
      cout << "YOYO\n";
    } else {
      cout << "NO\n";
    }
  }

  return 0;
}
```
